Make sure your kubectl client has >= 1.30 version

Install and Set Up kubectl

curl -L0 {{ .KubectlURI }} -o kubectl && \
    chmod +x kubectl && \
    sudo mv kubectl /usr/local/bin/ && \
    kubectl version --client

Install and Set Up kubelogin

curl -L0 {{ .KubeloginURI }} -o kubectl-oidc_login && \
    chmod +x kubectl-oidc_login && \
    sudo mv kubectl-oidc_login /usr/local/bin/ && \
    kubectl-oidc_login version

Copy this CA Certificate and download it to your .kube directory

curl --create-dirs -s {{ .IDPCaURI }} -o ${HOME}/.kube/certs/{{ .ClusterName }}/idp-ca.crt
{{ end }} {{ if .IDPCaPem }}

Run configuration commands

These commands will update ~/.kube/config

kubectl config set-cluster {{ .ClusterName }} \
{{- if or .K8sCaPem .K8sCaURI }}
--certificate-authority=${HOME}/.kube/certs/{{ .ClusterName}}/k8s-ca.crt \
{{- end }}
--server={{ .K8sMasterURI }}

Configure kubeconfig with kubelogin

kubectl config set-credentials {{ .Username }}-{{ .ClusterName }}-exec \
  --exec-api-version=client.authentication.k8s.io/v1 \
  --exec-interactive-mode=Never \
  --exec-command=kubectl \
  --exec-arg=oidc-login \
  --exec-arg=get-token \
  --exec-arg=--oidc-issuer-url={{ .Issuer }} \
  --exec-arg=--oidc-client-id={{ .ClientID }} \
  --exec-arg=--oidc-client-secret={{ .ClientSecret }} \
  --exec-arg=--oidc-extra-scope=email \
  --exec-arg=--oidc-extra-scope=profile \
  --exec-arg=--oidc-extra-scope=groups \
  --exec-arg=--oidc-extra-scope=openid \
  --exec-arg=--insecure-skip-tls-verify=true \
  --exec-arg=--grant-type=device-code \
  --exec-arg=--listen-address=0.0.0.0:443 \
  --exec-arg=--skip-open-browser=true \
  --exec-arg=--oidc-redirect-url-hostname={{ printf "%s/callback" .Issuer  }}
kubectl config set-context {{ .Username }}-{{ .ClusterName }}-exec \
  --cluster={{ .ClusterName }}{{ if .Namespace }} --namespace={{ .Namespace }}{{ end }} \
  --user={{ .Username}}-{{.ClusterName }}-exec
kubectl config use-context {{ .Username }}-{{ .ClusterName }}-exec
{{ end }}
