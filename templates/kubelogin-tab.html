Set up kubelogin credentials

Save the CA certificates:

{{ if .K8sCaURI }} 
curl --create-dirs -s {{ .K8sCaURI }} -o ${HOME}/.kube/certs/{{ .ClusterName }}/k8s-ca.crt
{{ end }} {{ if .IDPCaURI }} 
curl --create-dirs -s {{ .IDPCaURI }} -o ${HOME}/.kube/certs/{{ .ClusterName }}/idp-ca.crt
{{ end }}

Configure kubeconfig with kubelogin

kubectl config set-credentials {{ .Username }}-{{ .ClusterName }} \
  --exec-command=kubelogin \
  --exec-api-version=client.authentication.k8s.io/v1beta1 \
  --exec-arg=get-token \
  --exec-arg=--oidc-issuer-url={{ .Issuer }} \
  --exec-arg=--oidc-client-id={{ .ClientID }} \
  --exec-arg=--oidc-client-secret={{ .ClientSecret }} \
  --exec-arg=--grant-type=device_code
kubectl config set-context {{ .Username }}-{{ .ClusterName }} \
  --cluster={{ .ClusterName }}{{ if .Namespace }} --namespace={{ .Namespace }}{{ end }} \
  --user={{ .Username}}-{{.ClusterName }}
kubectl config use-context {{ .Username }}-{{ .ClusterName }}
{{ end }}
